```diff
- }
+ if ((child as THREE.Mesh).isMesh) {
+   const mat = (child as THREE.Mesh).material as THREE.Material & { opacity: number; _isAtm?: boolean; _isCloud?: boolean };
+   mat.opacity = mat._isAtm ? eA * 0.18 : (mat._isCloud ? eA * 0.28 : eA);
+ }
```

```diff
// Line 293-296 — cloud mesh onBeforeCompile
- onBeforeCompile={(shader) => {
-   (shader as any)._isCloud = true;
- }}
+ // Remove onBeforeCompile entirely. This doesn't work as intended —
+ // onBeforeCompile modifies the SHADER, not the MATERIAL.
+ // Instead, add a ref and tag the MATERIAL in a useEffect.
```

```diff
// Line 308-310 — atmosphere mesh onBeforeCompile
- onBeforeCompile={(shader) => {
-   (shader as any)._isAtm = true;
- }}
+ // Same fix: remove and use refs + useEffect to tag the material.
```

> [!IMPORTANT]
> **Root cause of Earth invisibility**: The `onBeforeCompile` sets `_isCloud`/`_isAtm` on the *shader object*, not the material. The traverse callback then checks `material._isAtm` which is always `undefined`. So every mesh gets `opacity = eA` (the main earth opacity), BUT the cloud and atmosphere are never tagged so they get incorrectly full opacity, which looks wrong AND the atmosphere BackSide material might occlude the earth.

**Correct fix — use refs for each material instead of traverse**:

Replace the entire `useFrame` Earth section (lines 199-216) with ref-based direct updates:

```typescript
// Add refs for each material at the top of the component:
const earthMeshRef = useRef<THREE.Mesh>(null!);
const cloudMatRef = useRef<THREE.MeshBasicMaterial>(null!);
const atmMatRef = useRef<THREE.MeshBasicMaterial>(null!);

// In useFrame:
const eA = ss(0.30, 0.44, s) * (1 - ss(0.52, 0.58, s));
if (eA > 0.001) {
  earthGroupRef.current.visible = true;
  // Direct ref updates — no traverse, no tagging
  (earthMeshRef.current.material as THREE.MeshPhongMaterial).opacity = eA;
  cloudMatRef.current.opacity = eA * 0.28;
  atmMatRef.current.opacity = eA * 0.18;
  // ... rest of rotation/scale
} else {
  earthGroupRef.current.visible = false;
}

// In JSX — add refs to each mesh/material:
<mesh ref={earthMeshRef}>
  <sphereGeometry args={[1.3, 64, 64]} />
  <meshPhongMaterial ... />
</mesh>
<mesh ref={cloudMeshRef} rotation={[0, 0.4, 0]}>
  <sphereGeometry args={[1.38, 32, 32]} />
  <meshBasicMaterial ref={cloudMatRef} ... />
</mesh>
<mesh>
  <sphereGeometry args={[1.46, 32, 32]} />
  <meshBasicMaterial ref={atmMatRef} ... />
</mesh>
```

**REMOVE** the `onBeforeCompile` callbacks entirely from both cloud and atmosphere meshes.

## 4. Transition Flare — Missing Entirely

### Root Cause

The transition flare (white flash between black hole and earth) was in the old `physova-hero.tsx` but was **never ported** to the R3F architecture. The `NarrativeScene.tsx` has no flare component.

### Fix

Create a new component or add the flare sprite to `CelestialBody.tsx`:

**In `CelestialBody.tsx`**, add to the JSX (inside the `<>` fragment, BEFORE the Earth group):

```tsx
{/* Transition Flare (BH → Earth) */}
<TransitionFlare />
```

**Add the TransitionFlare sub-component** (can be at the bottom of CelestialBody.tsx or a new file):

```tsx
const TransitionFlare = () => {
  const spriteRef = useRef<THREE.Sprite>(null!);
  const scrollProgress = useSimulationStore((state) => state.scrollProgress);
  
  const flareTexture = useMemo(() => {
    if (typeof document === 'undefined') return null;
    const c = document.createElement('canvas'); c.width = 128; c.height = 128;
    const ctx = c.getContext('2d')!;
    const g = ctx.createRadialGradient(64, 64, 0, 64, 64, 64);
    g.addColorStop(0, 'rgba(255,250,220,1)');
    g.addColorStop(0.15, 'rgba(255,220,140,0.8)');
    g.addColorStop(0.4, 'rgba(255,180,60,0.3)');
    g.addColorStop(1, 'rgba(255,120,20,0)');
    ctx.fillStyle = g; ctx.fillRect(0, 0, 128, 128);
    return new THREE.CanvasTexture(c);
  }, []);
  
  useFrame(() => {
    const s = scrollProgress;
    const flareA = ss(0.18, 0.24, s) * (1 - ss(0.30, 0.38, s));
    if (spriteRef.current.material instanceof THREE.SpriteMaterial) {
      spriteRef.current.material.opacity = flareA * 0.9;
    }
    if (flareA > 0.005) {
      const bloom = Math.pow(Math.sin(flareA * Math.PI), 0.6);
      spriteRef.current.scale.set(bloom * 5, bloom * 5, 1);
    } else {
      spriteRef.current.scale.set(0.01, 0.01, 1);
    }
  });
  
  if (!flareTexture) return null;
  
  return (
    <sprite ref={spriteRef}>
      <spriteMaterial
        map={flareTexture}
        transparent
        opacity={0}
        blending={THREE.AdditiveBlending}
        depthWrite={false}
      />
    </sprite>
  );
};
```

Also add a **DOM flash overlay** in `NarrativeWrapper.tsx` (or `SceneBase.tsx`):

Inside the `SceneBase` component div, add:

```tsx
{/* Transition flash overlay */}
<FlashOverlay />
```

Where `FlashOverlay` is:

```tsx
const FlashOverlay = () => {
  const flashRef = useRef<HTMLDivElement>(null!);
  const scrollProgress = useSimulationStore((state) => state.scrollProgress);
  
  useEffect(() => {
    // Update flash opacity on scroll
    const interval = setInterval(() => {
      const s = scrollProgress;
      const flashV = ss(0.18, 0.25, s) * (1 - ss(0.27, 0.36, s));
      if (flashRef.current) {
        flashRef.current.style.opacity = String(flashV * 0.22);
      }
    }, 16);
    return () => clearInterval(interval);
  });
  
  return (
    <div ref={flashRef} className="fixed inset-0 pointer-events-none z-[8]"
      style={{
        background: 'radial-gradient(ellipse 55% 40% at 50% 50%, rgba(255,245,210,0.85), rgba(220,190,90,0.25), transparent)',
        opacity: 0
      }}
    />
  );
};
```

> [!WARNING]
> The flash overlay needs to read `scrollProgress` from the Zustand store reactively. Since it's a DOM element (not R3F), you can't use `useFrame`. Use a `requestAnimationFrame` loop or `useEffect` with the store subscription instead. The `useSimulationStore` state value is just reading a number, which won't trigger re-renders. Instead, subscribe to the store directly:

```tsx
useEffect(() => {
  const unsub = useSimulationStore.subscribe((state) => {
    const s = state.scrollProgress;
    const flashV = ss(0.18, 0.25, s) * (1 - ss(0.27, 0.36, s));
    if (flashRef.current) {
      flashRef.current.style.opacity = String(flashV * 0.22);
    }
  });
  return unsub;
}, []);
```

## 5. Spacecraft Labels — Wrong DOM IDs + Style

### Root Cause

The `OverlayController` in `SceneBase.tsx` (lines 97-121) looks for an object named `'spacecraft'` using:

```typescript
const spacecraft = scene.getObjectByName('spacecraft');
```

The Spacecraft component **does** set `name="spacecraft"` on the group (line 135 of Spacecraft.tsx). So the name lookup is correct.

**However**, the `localToWorld` calls use hardcoded world-space offsets from the original engine where `planeGrp.scale` was `1.8`. In the new code, `shipScale` is conditionally `0.65` (mobile) or `1.8` (desktop), and the group position includes the `bentoLift`. The label anchor points `(1.5, 0, 0)` etc. are in LOCAL space of the spacecraft group, so `localToWorld` should still work IF the spacecraft is visible.

**The real problem**: the spacecraft's scroll range uses a DIFFERENT range than the label display range:

- Spacecraft visible: `ss(0.58, 0.66, s) * (1 - ss(0.74, 0.77, s))` — **fades out at 0.74-0.77**
- Labels display: `ss(0.67, 0.73, s) * (1 - ss(0.76, 0.82, s))` — **labels appear at 0.67 but ship is already fading at 0.74!**

This means labels only have a 0.67-0.74 window where both the ship AND labels are visible — **only 7% of scroll**. If the ship fades out too early, labels appear on an invisible object.

### Fix

Align the Spacecraft fade-out with the label range:

```diff
// Spacecraft.tsx line 92
- const planeA = ss(0.58, 0.66, s) * (1 - ss(0.74, 0.77, s));
+ const planeA = ss(0.58, 0.66, s) * (1 - ss(0.79, 0.85, s));
```

This matches the original HTML engine's range.

## 8. Starfield Performance

### Root Cause

In `Starfield.tsx`, the component uses `InstancedMesh` but **recalculates all 4200 instance matrices every single frame** (line 36-41):

```typescript
useFrame(() => {
  for (let i = 0; i < STAR_COUNT; i++) {
    tempObject.position.set(positions[i * 3], positions[i * 3 + 1], positions[i * 3 + 2]);
    tempObject.updateMatrix();
    meshRef.current.setMatrixAt(i, tempObject.matrix);
  }
  meshRef.current.instanceMatrix.needsUpdate = true;
});
```

Stars don't move! This is wasting massive CPU time.

### Fix

Move the position setup to an initialization effect, and only update opacity in the frame loop:

```diff
// Starfield.tsx — replace the useFrame with:

+ // Initialize matrices once
+ React.useEffect(() => {
+   if (!meshRef.current) return;
+   for (let i = 0; i < STAR_COUNT; i++) {
+     tempObject.position.set(positions[i * 3], positions[i * 3 + 1], positions[i * 3 + 2]);
+     tempObject.updateMatrix();
+     meshRef.current.setMatrixAt(i, tempObject.matrix);
+   }
+   meshRef.current.instanceMatrix.needsUpdate = true;
+ }, [positions, tempObject]);

  useFrame(() => {
    const s = scrollProgress;
    const flashV = ss(0.18, 0.25, s) * (1 - ss(0.27, 0.36, s));
    const opacity = 0.55 * (1 - flashV * 0.96);
    if (meshRef.current.material instanceof THREE.MeshBasicMaterial) {
```

## 9. Camera Choreography — Ultra-zoom unconditionally overwrites

### Root Cause

In `SceneBase.tsx` lines 35-37:

```typescript
const ultraZoomOut = ss(0.80, 0.90, s);
camZ = (shipCam > 0.01 ? 5.0 : camZ) + ultraZoomOut * 7.0;
camY = (shipCam > 0.01 ? camY : 0) + ultraZoomOut * 0.2;
```

When `s < 0.80`, `ultraZoomOut = 0`, so `camZ` gets set to either `5.0` (if shipCam is active) or `camZ` (from Earth zoom). But the code always evaluates `(shipCam > 0.01 ? 5.0 : camZ)` — and when `shipCam` is 0 (i.e., before spacecraft scene), `camZ` should remain at `6.5 - zoomInEarth * 1.5`. However it gets overwritten to `camZ + 0` which is fine.

**Actually the camera logic looks correct upon closer inspection.** The issue is that `shipCam` is `ss(0.62, 0.68, s) * (1 - ss(0.78, 0.84, s))`, and when outside the ship range, the `if (shipCam > 0.01)` branch doesn't execute. So `camZ` stays from the Earth zoom line.

**No change needed here** — the camera choreography is correct. The visual issues are caused by the other bugs documented above.
